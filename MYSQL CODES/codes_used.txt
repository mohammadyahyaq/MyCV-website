CREATE TABLE resumes (id INT(5) NOT NULL PRIMARY KEY AUTO_INCREMENT, first_name VARCHAR(15) NOT NULL, last_name VARCHAR(15) NOT NULL, major VARCHAR(40) NOT NULL, category VARCHAR(40) NOT NULL, page VARCHAR(50) NOT NULL);

CREATE TABLE feedbacks (
    ID INT(5) AUTO_INCREMENT PRIMARY KEY, fname VARCHAR(15) NOT NULL, lname VARCHAR(15) NOT NULL, email VARCHAR(40) NOT NULL, phone INT(10) NOT NULL, country VARCHAR(30) NOT NULL, city VARCHAR(30), info_source VARCHAR(8), is_easy VARCHAR(3), radio_easy VARCHAR(8), radio_rate VARCHAR(8), recommend_web VARCHAR(100), browse VARCHAR(100), subject VARCHAR(150) NOT NULL, suggestion VARCHAR(400)
);
